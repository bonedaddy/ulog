13 /*! @file logger.c
14  * @brief a thread safe logger with optional printf style logging
15  * @details allows writing color coded logs to stdout, with optional file output as
16  * well. timestamps all logs, and provides optional printf style logging
17  * @note logf_func has a bug where some format is respected and others are not,
18  * consider the following from a `%s%s` format:
19  * - [error - Jul 06 10:01:07 PM] one<insert-tab-here>two
20  * - [warn - Jul 06 10:01:07 PM] one two
21  * @note warn, and info appear to not respect format, while debug and error do
22  * @todo
23  * - buffer logs and use a dedicated thread for writing (avoid blocking locks)
24  * - handling system signals (exit, kill, etc...)
25  */

38 /*! @brief returns a new thread safe logger
39  * if with_debug is false, then all debug_log calls will be ignored
40  * @param with_debug whether to enable debug logging, if false debug log calls will
41  * be ignored
42  */
43 thread_logger *new_thread_logger(bool with_debug) {
44 
45  thread_logger *thl = malloc(sizeof(thread_logger));
46  if (thl == NULL) {
47  printf("failed to malloc thread_logger\n");
48  return NULL;
49  }
50 
51  thl->lock = pthread_mutex_lock;
52  thl->unlock = pthread_mutex_unlock;
53  thl->log = log_func;
54  thl->logf = logf_func;
55  thl->debug = with_debug;
56  pthread_mutex_init(&thl->mutex, NULL);
57 
58  return thl;
59 }

63  * @param output_file the file we will dump logs to. created if not exists and is

66 file_logger *new_file_logger(char *output_file, bool with_debug) {
67 
68  thread_logger *thl = new_thread_logger(with_debug);
69  if (thl == NULL) {
70  // dont printf log here since new_thread_logger handles that

75  if (fhl == NULL) {
76  // free thl as it is not null
77  free(thl);
78  printf("failed to malloc file_logger\n");
79  return NULL;
80  }
81 
82  // append to file, create if not exist, sync write files
83  // TODO(bonedaddy): try to use O_DSYNC for data integrity sync
84  int file_descriptor =
85  open(output_file, O_WRONLY | O_CREAT | O_SYNC | O_APPEND, 0640);
86  if (file_descriptor <= 0) {
87  // free thl as it is not null
88  free(thl);
89  // free fhl as it is not null
90  free(fhl);
91  printf("failed to run posix open function\n");
92  return NULL;
93  }
94 
95  fhl->fd = file_descriptor;
96  fhl->thl = thl;
97 
98  return fhl;
99 }

101 /*! @brief used to write a log message to file although this really means a file
102  * descriptor
103  * @param thl pointer to an instance of thread_logger
104  * @param file_descriptor file descriptor to write log messages to in addition to
105  * stdout logging. if 0 only stdout is used
106  * @param message the actuall message to log
107  */
108 int write_file_log(int file_descriptor, char *message) {
109 
110  char *msg = calloc(1, strlen(message) + 2); // 2 for \n
111  if (msg == NULL) {
112  printf("failed to calloc msg\n");
113  return -1;
114  }
115 
116  strcat(msg, message);
117  strcat(msg, "\n");
118 
119  int response = write(file_descriptor, msg, strlen(msg));
120  if (response == -1) {
121  printf("failed to write file log message");
122  } else {
123  // this branch will be triggered if write doesnt fail
124  // so overwrite the response to 0 as we want to return 0 to indicate
125  // no error was received, and returning response directly would return the
126  // number of bytes written
127  response = 0;
128  }
129 
130  free(msg);
131 
132  return response;
133 }

135 /*! @brief like log_func but for formatted logs
136  * @param thl pointer to an instance of thread_logger
137  * @param file_descriptor file descriptor to write log messages to, if 0 then only
138  * stdout is used
139  * @param level the log level to use (effects color used)
140  * @param message format string like `<percent-sign>sFOO<percent-sign>sBAR`
141  * @param ... values to supply to message
142  */
143 void logf_func(thread_logger *thl, int file_descriptor, LOG_LEVELS level, char *file,
144  int line, char *message, ...) {
145 
146  va_list args;
147  va_start(args, message);
148  char msg[sizeof(args) + (strlen(message) * 2)];
149  memset(msg, 0, sizeof(msg));
150 
151  int response = vsnprintf(msg, sizeof(msg), message, args);
152  if (response < 0) {
153  free(msg);
154  printf("failed to vsprintf\n");
155  return;
156  }
157 
158  log_func(thl, file_descriptor, msg, level, file, line);
159 }

161 /*! @brief main function you should call, which will delegate to the appopriate *_log
162  * function
163  * @param thl pointer to an instance of thread_logger
164  * @param file_descriptor file descriptor to write log messages to, if 0 then only
165  * stdout is used
166  * @param message the actual message we want to log
167  * @param level the log level to use (effects color used)
168  */
169 void log_func(thread_logger *thl, int file_descriptor, char *message,
170  LOG_LEVELS level, char *file, int line) {
171 
172  char *time_str = get_time_string();
173  if (time_str == NULL) {
174  // dont printf log as get_time_str does that
175  return;
176  }
177 
178  char location_info[strlen(file) + sizeof(line) + 4];
179  memset(location_info, 0, sizeof(location_info));
180 
181  sprintf(location_info, " %s:%i", file, line);
182 
183  char
184  date_msg[strlen(time_str) + strlen(message) + 2 + sizeof(location_info) + 6];
185  memset(date_msg, 0, sizeof(date_msg));
186 
187  strcat(date_msg, time_str);
188  strcat(date_msg, " -");
189  strcat(date_msg, location_info);
190  strcat(date_msg, "] ");
191  strcat(date_msg, message);
192 
193  switch (level) {
194  case LOG_LEVELS_INFO:
195  info_log(thl, file_descriptor, date_msg);
196  break;
197  case LOG_LEVELS_WARN:
198  warn_log(thl, file_descriptor, date_msg);
199  break;
200  case LOG_LEVELS_ERROR:
201  error_log(thl, file_descriptor, date_msg);
202  break;
203  case LOG_LEVELS_DEBUG:
204  debug_log(thl, file_descriptor, date_msg);
205  break;
206  }
207 
208  free(time_str);
209 }
